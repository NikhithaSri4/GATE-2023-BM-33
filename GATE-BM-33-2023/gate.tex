\textbf{Question:} 
A continuous time, band-limited signal $x(t)$ has its Fourier transform described by:\\
\begin{equation}
 X\brak{f} = \begin{cases} 
1 - \frac{\abs{f}}{200} & \text{if } \abs{f} \leq 200 \\
0 & \text{if } \abs{f} > 200 \label{eq:BM33}
\end{cases}  \\
\end{equation}
The signal is uniformly sampled at a sampling rate of 600 Hz. The Fourier transform of the signal is $X_s(f)$. What is the value of $\frac{X_s(600)}{X_s(500)}$? \\

\textbf{Solution: }

\begin{table}[h]
 	\centering
 	\resizebox{6 cm}{!}{
 		\input{tables/table.tex}
 	}
 	\caption{Input Parameters}
    \label{tab:table_33}
 \end{table}
 
\begin{align}
X_s\brak{f} &= \frac{1}{600} \sum_{k=-\infty}^{\infty} X\brak{f - 600k} \\
\implies X_s\brak{f+600} &= \frac{X(f)}{600}  \eqref{eq:BM33} \\
X_s\brak{600} = \frac{X\brak{0}}{600} &= \frac{1}{600} \\
X_s\brak{500} = \frac{X\brak{-100}}{600} &= \frac{1}{2\cdot 600} \\
\frac{X_s\brak{600}}{X_s\brak{500}} &= 2 
\end{align}
